```c
// 다음과 같은 코드를 이해하고 작성해봅시다.
#include<stdio.h>

int main() {
    int gugu[9][9];

    for(int i=0; i<9; i++) {
        for(int j=0; j<9; j++) {
            gugu[i][j] = (i+1) * (j+1);
        }
    }

    for(int i=0; i<9; i++) {
        for(int j=0; j<9; j++) {
            printf("%d * %d = %d\t", i+1, j+1, gugu[i][j]);
        }
        printf("\n");
    }

    return 0;
}
```
